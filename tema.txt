            <p style="font-size: 14px;">Drop <b>.cbz</b> / <b>.zip</b></p>

        <input type="file" id="fileInput" accept=".zip,.cbz">

    <script>
        const fileInput = document.getElementById('fileInput');
        const status = document.getElementById('status');
        const viewer = document.getElementById('viewer');
        const setup = document.getElementById('setup');
        const header = document.getElementById('mainHeader');
        const topControls = document.getElementById('topControls');

        // Logik Auto-Hide Header
        let lastScroll = 0;
        window.addEventListener('scroll', () => {
            const currentScroll = window.pageYOffset;
            if (currentScroll > lastScroll && currentScroll > 50) {
                header.classList.add('hide');
            } else {
                header.classList.remove('hide');
            }
            lastScroll = currentScroll;
        });

        fileInput.onchange = (e) => handleFile(e.target.files[0]);

        async function handleFile(file) {
            if (!file) return;
            status.style.display = 'block';

            try {
                const zip = await JSZip.loadAsync(file);
                const files = [];

                zip.forEach((path, entry) => {
                    if (entry.name.match(/\.(jpg|jpeg|png|webp|gif)$/i) && !entry.name.includes('__MACOSX')) {
                        files.push(entry);
                    }
                });

                files.sort((a, b) => a.name.localeCompare(b.name, undefined, {numeric: true}));

                for (let fileObj of files) {
                    const blob = await fileObj.async("blob");
                    const img = document.createElement('img');
                    img.src = URL.createObjectURL(blob);
                    viewer.appendChild(img);
                }

                setup.style.display = 'none';
                topControls.style.display = 'block';
                window.scrollTo(0, 0);

            } catch (err) {
                alert("Error: " + err);
                location.reload();
            }
        }

        // Drag & Drop support
        const dropZone = document.querySelector('.drop-zone');
        dropZone.ondragover = (e) => e.preventDefault();
        dropZone.ondrop = (e) => { e.preventDefault(); handleFile(e.dataTransfer.files[0]); };
    </script>
